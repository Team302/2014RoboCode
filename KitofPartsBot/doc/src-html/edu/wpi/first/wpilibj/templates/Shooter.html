003package edu.wpi.first.wpilibj.templates;
004
005import edu.wpi.first.wpilibj.DoubleSolenoid;
006import edu.wpi.first.wpilibj.Solenoid;
007import edu.wpi.first.wpilibj.Talon;
008import edu.wpi.first.wpilibj.Joystick;
009import edu.wpi.first.wpilibj.Encoder;
010
011/**
012 *
013 * @author John
014 */
015public class Shooter {
016    
017   public Shooter(DoubleSolenoid jaws, Solenoid shooterholder, Talon shootermotor, Encoder shooterencoder, Joystick coopstick, int shooteroverride) {
018       Jaws = jaws;
019       ShooterHolder = shooterholder;
020       ShooterMotor = shootermotor;
021       ShooterEncoder = shooterencoder;
022       CoOpstick = coopstick;
023       SHOOTER_OVERRIDE = shooteroverride;
024   }
025
026    public void init() {
027        ShooterState = SHOOTER_STOP;
028    }
029
030    // call this to start the shooter sequence
031    public void trigger() {
032        if (ShooterState == SHOOTER_STOP && Jaws.get() == DoubleSolenoid.Value.kForward) {
033            ShooterState = SHOOTER_PREPARE;
034            ShooterTimer = 0;
035            ShooterHolder.set(true);
036        }
037        // else its already in motion
038    }
039
040    // this gets called every 20ms in both Autonomous and Teleop
041    public void periodic() {
042        //  Timeout to stop it in case it gets stuck
043        ShooterTimer++;
044        if (ShooterTimer > 200) {   // give up after 4 sec.
045            ShooterState = SHOOTER_STOP;
046        }
047
048        //  Take action depending on the state
049        switch (ShooterState) {
050
051            case SHOOTER_STOP:
052                ShooterMotor.set(0);
053                ShooterHolder.set(false);
054                ShooterTimer = 0;
055                break;
056
057            case SHOOTER_PREPARE:
058                if (CoOpstick.getRawButton(SHOOTER_OVERRIDE)) {
059                    ShooterState = SHOOTER_MANUAL;
060                } else {
061                    ShooterState = SHOOTER_RETRACT;
062                }
063                break;
064
065            case SHOOTER_RETRACT:
066                if (ShooterEncoder.getDistance() > SHOOTER_BACKSTOP) {
067                    ShooterMotor.set(-1);   // pull back
068                } else {
069                    ShooterMotor.set(0);    // stop for 20ms
070                    ShooterState = SHOOTER_FIRE;
071                }
072                break;
073
074            case SHOOTER_FIRE:
075                if (ShooterEncoder.getDistance() < SHOOTER_FORWARDSTOP) {
076                    ShooterMotor.set(1);    // swing to shoot
077                } else {
078                    ShooterMotor.set(0);    // stop
079                    ShooterState = SHOOTER_RELAX;
080                }
081                break;
082
083            case SHOOTER_MANUAL:
084                if (ShooterTimer < 25) {
085                    ShooterMotor.set(-.4);
086                } else {
087                    if (ShooterTimer < 75) {
088                        ShooterMotor.set(1);
089                    } else {
090                        if (ShooterTimer < 100) {
091                        ShooterMotor.set(-.4);
092                        } else {
093                            ShooterState = SHOOTER_STOP;
094                        }
095                    }
096                }
097                break;
098
099            case SHOOTER_RELAX:
100                if (ShooterEncoder.getDistance() < SHOOTER_SAFEANGLE) {
101                    ShooterState = SHOOTER_STOP;
102                } else {
103                    ShooterMotor.set(-0.400);
104                }
105                break;
106
107            default:
108                ShooterMotor.set(0);
109                ShooterState = SHOOTER_STOP;
110                break;
111        }
112        SmartDashboardData.putNumber("Shooter Encoder", ShooterEncoder.getDistance());
113    }
114    
115    private double leftencodercount;
116    private double rightencodercount;
117    private int ShooterTimer;
118    private int ShooterState;
119    private final int SHOOTER_STOP = 0;
120    private final int SHOOTER_PREPARE = 1;
121    private final int SHOOTER_RETRACT = 2;
122    private final int SHOOTER_FIRE = 3;
123    private final int SHOOTER_RETURN = 4;
124    private final int SHOOTER_RELAX = 5;
125    private final int SHOOTER_MANUAL = 6;
126
127    // encoder positions
128    private final double SHOOTER_BACKSTOP = -60;     // stop retracting when angle is less than this
129    private final double SHOOTER_SAFEANGLE = -10;     // move backward until the shooter greater than this
130    private final double SHOOTER_FORWARDSTOP = 70;      // stop forward motion here
131    
132    private final DoubleSolenoid Jaws;
133    private final Solenoid ShooterHolder;
134    private final Talon ShooterMotor;
135    private final Encoder ShooterEncoder;
136    private final Joystick CoOpstick;
137    private final int SHOOTER_OVERRIDE;
138}
